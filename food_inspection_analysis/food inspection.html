    header_names = []

        for header in headers:
            header_names.append(header.text)
            if header.text == 'Name':

        row_content = row.find_all('td')
        for val in row_content:
            single_row_data.append(val.text)

                hyperlink = val.find('a')
                single_row_data.append(hyperlink['href'])
        if single_row_data:
